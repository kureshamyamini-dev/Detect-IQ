<script>
document.getElementById('claimForm').addEventListener('submit', async (e) => {
    e.preventDefault();
    const btn = e.target.querySelector('button');
    const originalText = btn.innerHTML;
    btn.innerHTML = 'Analyzing...';
    btn.disabled = true;

    const payload = {
        claim_amount: document.getElementById('claimAmount').value,
        customer_age: document.getElementById('customerAge').value,
        months_as_customer: document.getElementById('monthsCustomer').value,
        previous_claims: document.getElementById('prevClaims').value,
        policy_deductable: document.getElementById('deductible').value,
        incident_severity: document.getElementById('severity').value,
        model: document.getElementById('modelChoice').value
    };

    try {
        const response = await fetch('/api/predict', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify(payload)
        });
        const data = await response.json();
        
        const resultBox = document.getElementById('resultBox');
        const riskScore = document.getElementById('riskScore');
        const resultLabel = document.getElementById('resultLabel');
        const resultMessage = document.getElementById('resultMessage');
        
        resultBox.style.display = 'block';
        riskScore.textContent = data.fraud_risk_score + '%';
        
        if (data.is_fraudulent) {
            riskScore.className = 'text-danger';
            resultLabel.textContent = 'High Risk - Investigation Required';
            resultLabel.className = 'text-danger';
            resultBox.style.borderColor = 'rgba(239, 68, 68, 0.4)';
            resultBox.style.background = 'rgba(239, 68, 68, 0.05)';
            resultMessage.innerHTML = `Model <strong>${data.model_used}</strong> flagged this claim as highly suspicious based on historical patterns. Recommended action: Manual review team and SIU referral.`;
        } else {
            riskScore.className = 'text-success';
            resultLabel.textContent = 'Low Risk - Genuine Claim';
            resultLabel.className = 'text-success';
            resultBox.style.borderColor = 'rgba(16, 185, 129, 0.4)';
            resultBox.style.background = 'rgba(16, 185, 129, 0.05)';
            resultMessage.innerHTML = `Model <strong>${data.model_used}</strong> indicates normal patterns. Safe to proceed with standard processing.`;
        }
    } catch(err) {
        alert('Error communicating with AI models. Ensure backend is running.');
    } finally {
        btn.innerHTML = originalText;
        btn.disabled = false;
    }
});
</script>
